基本使用

创建表、插入数据、查询数据等基本操作

import { createTable, insert, findOne, findMany, update, remove } from 'expo-lite-db-store';

// 创建表
await createTable('users');

// 插入数据
await insert('users', [
  { id: 1, name: '张三', email: 'zhangsan@example.com', age: 25, active: true },
  { id: 2, name: '李四', email: 'lisi@example.com', age: 30, active: false },
  { id: 3, name: '王五', email: 'wangwu@example.com', age: 35, active: true }
]);

// 查询单条数据
const user = await findOne('users', { id: 1 });
console.log(user); // { id: 1, name: '张三', ... }

// 查询多条数据
const activeUsers = await findMany('users', { active: true });
console.log(activeUsers.length); // 2

// 更新数据
await update('users', { age: 26 }, { id: 1 });

// 删除数据
await remove('users', { id: 2 });

// 查询所有数据
const allUsers = await findMany('users');
console.log(allUsers.length); // 2

事务操作

使用事务确保数据一致性

import { beginTransaction, commit, rollback, insert, update } from 'expo-lite-db-store';

try {
  // 开始事务
  await beginTransaction();

  // 执行多个操作
  await insert('orders', { id: 1, userId: 1, amount: 100, status: 'pending' });
  await update('users', { balance: 900 }, { id: 1 });

  // 提交事务
  await commit();
  console.log('Transaction committed successfully');
} catch (error) {
  // 回滚事务
  await rollback();
  console.error('Transaction failed:', error);
}

复杂查询

使用各种查询条件和操作符

import { findMany } from 'expo-lite-db-store';

// 使用比较操作符
const users = await findMany('users', {
  age: { $gt: 25 }, // 年龄大于 25
  active: true
});

// 使用逻辑操作符
const filteredUsers = await findMany('users', {
  $or: [
    { age: { $lt: 25 } },
    { active: false }
  ]
});

// 使用 in 操作符
const specificUsers = await findMany('users', {
  id: { $in: [1, 3, 5] }
});

// 使用分页和排序
const paginatedUsers = await findMany('users',
  { active: true },
  {
    skip: 10,  // 跳过前 10 条
    limit: 20, // 返回 20 条
    sort: { createdAt: -1 } // 按创建时间降序排序
  }
);

加密存储

使用加密存储保护敏感数据

import { createTable, insert, findOne } from 'expo-lite-db-store';

// 创建加密表
await createTable('sensitive_data', {
  encrypted: true,
  encryptionKey: 'your-secret-key-123456' // 密钥由你完全控制
});

// 插入敏感数据
await insert('sensitive_data', {
  userId: 1,
  creditCard: '4111111111111111',
  cvv: '123',
  expiryDate: '12/25'
});

// 查询加密数据
const data = await findOne('sensitive_data', { userId: 1 });
console.log(data); // 自动解密后的数据

批量操作

使用批量操作提高性能

import { bulkWrite } from 'expo-lite-db-store';

// 批量写入操作
await bulkWrite('users', [
  { type: 'insert', data: { id: 4, name: '赵六', email: 'zhaoliu@example.com' } },
  { type: 'insert', data: { id: 5, name: '孙七', email: 'sunqi@example.com' } },
  { type: 'update', where: { id: 1 }, data: { age: 27 } },
  { type: 'remove', where: { id: 2 } }
]);

// 批量插入大量数据
const largeData = Array.from({ length: 1000 }, (_, i) => ({
  id: i + 1000,
  name: `User ${i + 1000}`,
  email: `user${i + 1000}@example.com`,
  age: 20 + (i % 50)
}));

await bulkWrite('users', largeData.map(data => ({
  type: 'insert',
  data
})));

用户管理

import { createTable, insert, findOne, update, remove } from 'expo-lite-db-store';

// 创建用户表
await createTable('users', {
  indexes: ['email', 'username'],
  autoIndex: true
});

// 注册用户
async function registerUser(userData) {
  // 检查邮箱是否已存在
  const existingUser = await findOne('users', { email: userData.email });
  if (existingUser) {
    throw new Error('Email already exists');
  }

  // 插入新用户
  const result = await insert('users', userData);
  return result;
}

  return user;

待办事项应用

import { createTable, insert, findMany, update, remove } from 'expo-lite-db-store';

// 创建待办事项表
await createTable('todos', {
  indexes: ['userId', 'status'],
  autoIndex: true
});

// 添加待办事项
async function addTodo(userId, todoData) {
  const todo = {
    ...todoData,
    userId,
    status: 'pending',
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString()
  };

  const result = await insert('todos', todo);
  return result;
}

// 获取用户的待办事项
async function getTodos(userId, filters = {}) {
  const todos = await findMany('todos', {
    userId,
    ...filters
  }, {
    sort: { createdAt: -1 }
  });

  return todos;
}

// 更新待办事项状态
async function updateTodoStatus(todoId, status) {
  const result = await update('todos', {
    status,
    updatedAt: new Date().toISOString()
  }, { id: todoId });

  return result;
}

// 删除待办事项
async function deleteTodo(todoId) {
  const result = await remove('todos', { id: todoId });
  return result;
}

// 批量更新待办事项
async function batchUpdateTodos(todoIds, updates) {
  const operations = todoIds.map(id => ({
    type: 'update',
    where: { id },
    data: {
      ...updates,
      updatedAt: new Date().toISOString()
    }
  }));

  const result = await bulkWrite('todos', operations);
  return result;
}
