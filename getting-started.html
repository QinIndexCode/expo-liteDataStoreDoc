入门指南

开始使用 expo-liteDBStore

快速上手轻量级本地存储解决方案，为 React Native + Expo 项目设计。

前提条件

  • 1
    React Native 项目
  • 2
    Expo SDK 49+
  • 3
    Node.js 16+

安装指南

安装步骤

1 安装依赖 

npm install expo-lite-db-store

或者使用 yarn 或 pnpm:

yarn add expo-lite-db-store
# 或者
pnpm add expo-lite-db-store

2 配置项目

你可以创建一个配置文件来自定义 expo-liteDBStore 的行为：

// src/liteStore.config.js
module.exports = {
  // 存储路径
  storagePath: 'expo-liteDBStore',
  // 单文件最大大小（字节）
  maxFileSize: 5 * 1024 * 1024, // 5MB
  // 默认加密密钥（可选）
  encryptionKey: 'your-secret-key',
  // 缓存大小
  cacheSize: 1000,
  // 日志级别
  logLevel: 'info'
};

3 开始使用

现在你可以在你的项目中使用 expo-liteDBStore 了！

基本使用

创建表 

import { createTable } from 'expo-lite-db-store';

// 创建一个名为 users 的表
await createTable('users');

// 或者使用选项创建表
await createTable('users', {
  // 是否启用加密
  encrypted: true,
  // 是否自动创建索引
  autoIndex: true,
  // 索引字段
  indexes: ['id', 'email']
});

插入数据 

import { insert } from 'expo-lite-db-store';

// 插入单条数据
await insert('users', { id: 1, name: '张三', email: 'zhangsan@example.com', age: 25 });

// 插入多条数据
await insert('users', [
  { id: 2, name: '李四', email: 'lisi@example.com', age: 30 },
  { id: 3, name: '王五', email: 'wangwu@example.com', age: 35 }
]);

查询数据 

import { findOne, findMany } from 'expo-lite-db-store';

// 查询单条数据
const user = await findOne('users', { id: 1 });
console.log(user); // { id: 1, name: '张三', email: 'zhangsan@example.com', age: 25 }

// 查询多条数据
const users = await findMany('users', { age: { $gt: 25 } });
console.log(users); // 所有年龄大于 25 的用户

// 使用选项查询
const paginatedUsers = await findMany('users',
  { active: true },
  {
    skip: 10,  // 跳过前 10 条
    limit: 20, // 返回 20 条
    sort: { createdAt: -1 } // 按创建时间降序排序
  }
);

更新数据 

import { update } from 'expo-lite-db-store';

// 更新单条数据
const updatedCount = await update('users', { age: 26 }, { id: 1 });
console.log(updatedCount); // 1

// 更新多条数据
const updatedCount = await update('users', { active: true }, { age: { $gt: 30 } });
console.log(updatedCount); // 更新的记录数

删除数据 

import { remove } from 'expo-lite-db-store';

// 删除单条数据
const deletedCount = await remove('users', { id: 1 });
console.log(deletedCount); // 1

// 删除多条数据
const deletedCount = await remove('users', { age: { $lt: 18 } });
console.log(deletedCount); // 删除的记录数
